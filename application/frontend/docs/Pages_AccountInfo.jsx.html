import React from "react";
import { Link, useNavigate } from "react-router-dom";
import { Form, Button } from "react-bootstrap";

/**
 * The account info editing page. 
 * With a form that allows users to edit their email and password.
 * It hadn't been wired with the backend yet.
 * @returns {ReactNode} The account info editing page.
 */

function AccountInfo() {
  const navigate = useNavigate();

  return (
    <Form>
      <table className="table">
        <tbody>
          <tr>
            <td scope="row">
              <Form.Label>EMAIL</Form.Label>
            </td>
            <td className="w-75">
              <Form.Group controlId="formBasicUserName">
                <Form.Control type="text" placeholder="Username" />
              </Form.Group>
            </td>
          </tr>
          <tr>
            <td scope="row">
              <Form.Label>PASSWORD</Form.Label>
            </td>
            <td className="w-75">
              <Form.Group controlId="formBasicPassword">
                <Form.Control type="password" placeholder="Password" />
              </Form.Group>
            </td>
          </tr>
        </tbody>
      </table>
      <div className="ml-2">
        <Button
          className="btn btn-light mr-5"
          type="cancel"
          onClick={() => {
            navigate("/profile");
          }}
        >
          Cancel
        </Button>
        <Button type="button" class="btn btn-primary ml-5">
          Save
        </Button>
      </div>
    </Form>
  );
}
export default AccountInfo;

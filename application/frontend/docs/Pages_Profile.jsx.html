import { Button, Row, Col, Container } from "react-bootstrap";
import React from "react";
import { Link, useNavigate } from "react-router-dom";
/**
 * The profile page with links to edit account info, personal info, and set schedule preferences.
 * Comes with a logout button and a display of the user's ID.
 * @returns {ReactNode} The profile page.
 */
function Profile() {
  const navigate = useNavigate();
  const logout = () => {
    localStorage.clear();
    navigate("/landing");
  };

  return (
    <div className="Profile">
      <table className="table table-dark">
        <Container fluid="md">
          <Row>
            <Col>
              <Link
                className="text-decoration-none"
                style={{ color: "white" }}
                to="AccountInfo"
              >
                Account Info
              </Link>
            </Col>
          </Row>
          <Row>
            <Col>
              <Link
                className="text-decoration-none"
                style={{ color: "white" }}
                to="PersonalInfo"
              >
                Personal Info
              </Link>
            </Col>
          </Row>
          <Row>
            <Col>
              <Link
                className="text-decoration-none"
                style={{ color: "white" }}
                to="SchedulePref"
              >
                Schedule Preferences
              </Link>
            </Col>
          </Row>
        </Container>
      </table>

      <Button variant="danger w-100 mt-5 mb-3" onClick={logout}>
        Logout
      </Button>

      <div className="UID">UserID: {localStorage.userID}</div>
    </div>
  );
}
export default Profile;

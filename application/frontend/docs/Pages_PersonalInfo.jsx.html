import React from "react";
import { Form, Button } from "react-bootstrap";
import { useNavigate } from "react-router-dom";

/**
 * The personal info editing page.
 * With a form that allows users to edit their name, email, age and gender.
 * It hadn't been wired with the backend yet.
 * @returns {ReactNode} The personal info editing page.
 */

function PersonalInfo() {
  const navigate = useNavigate();

  return (
    <div className="personalInfo">
      <Form>
        <table className="table">
          <tbody>
            <tr>
              <td scope="row">
                <Form.Label>NAME</Form.Label>
              </td>
              <td className="w-75">
                <Form.Group controlId="formBasicUserName">
                  <Form.Control type="text" placeholder="Username" />
                </Form.Group>
              </td>
            </tr>
            <tr>
              <td scope="row">
                <Form.Label>EMAIL</Form.Label>
              </td>
              <td className="w-75">
                <Form.Group controlId="formBasicEmail">
                  <Form.Control type="email" placeholder="Email" />
                </Form.Group>
              </td>
            </tr>
            <tr>
              <td scope="row">
                <Form.Label>AGE</Form.Label>
              </td>
              <td className="w-75">
                <Form.Group controlId="formBasicAge">
                  <Form.Control type="number" placeholder="Age" />
                </Form.Group>
              </td>
            </tr>
            <tr>
              <td scope="row">
                <Form.Label>GENDER</Form.Label>
              </td>
              <td className="w-75">
                <Form.Group controlId="formBasicGender">
                  <Form.Control type="gender" placeholder="Gender" />
                </Form.Group>
              </td>
            </tr>
          </tbody>
        </table>
        <div className="btn-toolbar">
          <Button
            className="btn btn-light mx-3"
            type="cancel"
            onClick={() => {
              navigate("/profile");
            }}
          >
            Cancel
          </Button>
          <Button type="button" class="btn btn-primary">
            Save
          </Button>
        </div>
      </Form>
    </div>
  );
}
export default PersonalInfo;
